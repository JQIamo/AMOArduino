234  void insertToSetList(int pos, GenericSetListCallback function, int * params){
235 
236  // check that line you are appending is different from previous
237  bool eq = false;
238  if ((pos != 0) && (_setlist[pos - 1].function ==
239  reinterpret_cast<SpecificSetListCallback>(function))){
240  eq = true;
241 
242  // now check if params are equal..
243  for(int i = 0; i < MAX_PARAM_NUM; i++){
244  if(_setlist[pos - 1].params[i] != params[i]){
245  eq = false;
246  break;
247  }
248  }
249  }
250  if (!eq) {
251  _setlist[pos].function = reinterpret_cast
252  <SpecificSetListCallback>(function);
253  } else {
254  _setlist[pos].function = _holdValue;
255  }
256  // copy param list into SetListCallback...
257  memcpy(_setlist[pos].params, params, sizeof(params)*MAX_PARAM_NUM);
258  // increment setlistLength by 1
259 
260  _setlistLength++;
261  }
